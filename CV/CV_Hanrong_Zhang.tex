\section{Research Experience}
  \resumeSubHeadingListStart

    \resumeProjectHeading
      {\textbf{Agent Security Bench (ASB)}}{\textcolor{blue}{ICLR 2025, First Author}}
      \resumeItemListStart
        % \resumeItem{\textbf{Impact:} Now a \textbf{foundational and widely-adopted benchmark} for LLM-agent security, with \textbf{260+ GitHub stars} and \textbf{280+ citations} (as of Jun. 2026).}
        \resumeItem{\textbf{Motivation:} LLM agents extend LLM reasoning into real-world interaction: by calling tools, updating memory, and executing multi-step plans, they can change external environments, making agent security more critical than standard chatbot safety; however, prior evaluations lacked a unified benchmark across prompts, tools, memory, and planning.}
        \resumeItem{\textbf{Contribution:} Introduce the first LLM agent security benchmark with 10 scenarios, 10 agents, 400+ tools, 27 attack/defense methods, 7 metrics, and Net Resilient Performance (NRP) for balancing utility and security.}
        \resumeItem{\textbf{Results:} Reveal that current LLM agents remain highly vulnerable and existing defenses provide limited protection: attacks reach up to 84.30\% success across prompt, tool, and memory stages, while defenses struggle to preserve both utility and security across 13 LLM backbones and 90,000+ test cases.}
      \resumeItemListEnd

    \resumeProjectHeading
      {\textbf{Towards Self-Evolving Skills in LLM Agents}}{\textcolor{blue}{CoEvoSkills \& OpenSkill, Preprint}}
      \resumeItemListStart
        \resumeItem{\textbf{Motivation:} Skills can boost professional tasks, but human-curated skills are costly and can even hurt performance in some SkillsBench domains. Meanwhile, one-shot self-generation methods are nearly useless, and prior evolution methods are mostly designed for tools and cannot reliably build structured multi-file skill packages.}
        \resumeItem{\textbf{Contribution:} Propose \textbf{CoEvoSkills} for closed-world skill evolution and \textbf{OpenSkill} for open-world self-evolution, refining multi-file skills through surrogate verification, open-world knowledge retrieval.}
        \resumeItem{\textbf{Results:} \textbf{CoEvoSkills} reaches average 71.1\% Pass rate on SkillsBench (+40.5pp over no-skill, +17.6pp over human-curated skills baselines); \textbf{OpenSkill} reaches 43.6\% on Claude Opus 4.6 and 42.1\% on GPT 5.2, within 1--3pp of human-curated skills and about +9pp over the best closed-world baselines.}
      \resumeItemListEnd

    \resumeProjectHeading
      {\textbf{GAM: Hierarchical Graph-based Memory for LLM Agents}}{\textcolor{blue}{ACL 2026, Co-First \& Corresponding Author}}
      \resumeItemListStart
        \resumeItem{\textbf{Motivation:} Long-term agent memory faces a core conflict between fast contextual adaptation and stable knowledge retention: stream-based memories update quickly but lack write isolation, causing memory loss and semantic drift, while static graph memories preserve structure but struggle with fluid open-domain dialogue.}
        \resumeItem{\textbf{Contribution:} Propose \textbf{GAM}, a hierarchical graph memory framework that decouples memory encoding from consolidation: an Event Progression Graph isolates ongoing dialogue events, while semantic-boundary triggers consolidate complete episodes into a Topic Associative Network for stable long-term knowledge.}
        \resumeItem{\textbf{Results:} \textbf{GAM} achieves SOTA performance among six memory baselines on LoCoMo and LongDialQA.}
      \resumeItemListEnd
